Exercise 9

years <- page %>%
  html_nodes(".cli-title-metadata span:nth-child(1)") %>%
  html_text() %>%
  as.numeric()

print(head(years, 10))
##  [1] 1994 1972 2008 1974 1957 1993 2003 1994 2001 1966

Exercise 10

ratings <- page %>%
  html_nodes(".ipc-rating-star--imdb") %>%
  html_attr("aria-label") %>%
  str_sub(-3, -1) %>%
  as.numeric()

print(head(ratings, 10))
##  [1] 9.3 9.2 9.0 9.0 9.0 9.0 9.0 8.9 8.9 8.8

Exercise 11

imdb_top_250 <- data.frame(
  title = titles,
  year = years,
  rating = ratings
)
